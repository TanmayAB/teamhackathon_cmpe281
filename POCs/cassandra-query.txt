CREATE KEYSPACE starbucks
WITH replication = {'class':'SimpleStrategy', 'replication_factor' : 1};

USE starbucks;

CREATE TYPE order_items (
   qty int,
   name text,
   milk_type text,
   size text,
   price decimal
   );
   
   CREATE TABLE starbucks_order (
   order_id uuid PRIMARY KEY,
   amount decimal,
   location text,
   items list<frozen<order_items>>,
   status text,
   message text
   );
